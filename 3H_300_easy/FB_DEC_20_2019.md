200. Number of Islands
https://leetcode.com/problems/number-of-islands/

class Solution:
    def numIslands(self, grid: List[List[str]]) -> int:
        def helper(i, j):
            check = [[i, j]]
            while check:
                temp = []
                for a, b in check: 
                    if 0 < a: 
                        if grid[a-1][b] == "1":
                            grid[a-1][b] = 0
                            temp.append([a-1, b]) 
                    if a < n - 1:  
                        if grid[a+1][b] == "1":
                            grid[a+1][b] = 0
                            temp.append([a+1, b])
                    if 0 < b: 
                        if grid[a][b-1] == "1":
                            grid[a][b-1] = 0
                            temp.append([a, b-1]) 
                    if b < m - 1:  
                        if grid[a][b+1] == "1":
                            grid[a][b+1] = 0
                            temp.append([a, b+1])
                check = temp
            
            
        if not grid:
            return 0
        
        n, m = len(grid), len(grid[0])
        c = 0
        for i in range(n):
            for j in range(m):
                if grid[i][j] == "1":
                    c += 1 
                    grid[i][j] = "0"
                    helper(i, j)
        return c

# Definition for a binary tree node.
# class TreeNode:
#     def __init__(self, x):
#         self.val = x
#         self.left = None
#         self.right = None

class Solution:
    def isValidBST(self, root: TreeNode) -> bool:
        
        def helper(root):
            if not root:
                return 
            helper(root.left)
            stack.append(root.val)
            helper(root.right) 
        
        stack = []
        helper(root) 
        if not stack:
            return True
        prev = stack[0]
        for i in stack[1:]:
            if i > prev:
                prev = i
            else:
                return False 
        return True
