class Solution:
    def flatten(self, root: TreeNode) -> None:
        """
        Do not return anything, modify root in-place instead.
        """
        def pre_order(root):
            if not root:
                return 
            lst.append(root.val)
            pre_order(root.left)
            pre_order(root.right) 
        lst = []
        pre_order(root)
        while lst:
            root.val = lst.pop(0)
            root.left = None
            if lst:
                root.right = TreeNode(0)
                root = root.right
            else:
                root.right = None
