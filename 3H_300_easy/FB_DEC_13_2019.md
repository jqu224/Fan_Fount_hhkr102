class Solution:
    def numDecodings(self, s: str) -> int:
        if s[0] == "0":
            return 0
        prev = ""
        memo = [1]
        
        for i in s:
            if prev and i == "0":
                memo.append(memo[-2])
            elif prev == i == "0":
                return 0
            elif prev and prev in "12" and i in "0123456": 
                memo.append(memo[-2] + memo[-1])
            else:
                memo.append(memo[-1])
            prev = i 
        return memo[-1]
                
                
                
class Solution:
    def numDecodings(self, s: str) -> int:
        if len(s) == 0:
            return 0
        if s[0] == "0":
            return 0
        dp = [0] * (len(s) + 1)
        dp[0] = 1
        dp[1] = 1
        for i in range(2, len(s)+1):
            number = int(s[i-1])
            # print(number)
            nb_with_prev = int(s[i-2:i])
            # print(nb_with_prev)
            if nb_with_prev <= 26 and nb_with_prev > 9:
                dp[i] += dp[i-2]
            if number > 0:
                dp[i] += dp[i-1]
            if dp[i] == 0:
                return 0
        return dp[-1]
            
class Solution:
    def numDecodings(self, s: str) -> int:
        dp = [1 for _ in range(len(s)+1)]
        if s[0] == '0': # edge cases, if the string starts with "0", we return 0
            return 0
        for i in range(1, len(s)):
            if s[i] != '0' and s[i-1] != "0":  # if the string is "normal"
                if 1 <= int(s[i-1:i+1]) <= 26: # if the two digits are within the covered range
                    dp[i+1] = dp[i] + dp[i-1]
                else:
                    dp[i+1] = dp[i] # the last step can only be 1 step away
            else:
                if int(s[i-1:i+1]) == 0 or int(s[i-1:i+1]) > 26: # edge cases like "00" or "30"
                    return 0
                else:
                    dp[i+1] = dp[i-1]  # cases like "10", we need to go two steps back, meaning that we can only decode "10" as a whole.
            
        return dp[-1]

15. 3Sum
https://leetcode.com/problems/3sum/submissions/
```
from collections import defaultdict
class Solution:
    def threeSum(self, nums: List[int]) -> List[List[int]]:
        dd = defaultdict(int)
        for i in nums:
            dd[i] += 1
        new_list = sorted(dd.keys())
        ret = []
        for i, a in enumerate(new_list):
            for b in new_list[i:]:
                if a == b and dd[a] == 1:
                    continue
                c = 0 - a - b  
                if c in dd and c >= b:
                    if c == 0 and dd[0] < 3:
                        continue
                    if c == b and dd[c] == 1:
                        continue
                    ret.append([a,b,c])
        return ret

class Solution:
    def threeSum(self, nums: List[int]) -> List[List[int]]:
        nums = sorted(nums)
        result = set()
        for i in range(len(nums)):
            l = i + 1
            r = len(nums) - 1
            target = 0 - nums[i]
            while l < r:
                if nums[l] + nums[r] == target:
                    result.add((nums[i], nums[l], nums[r]))
                    l += 1
                    r -= 1
                elif nums[l] + nums[r] < target:
                    l += 1
                else:
                    r -= 1
        return list(result)

29. Divide Two Integers
https://leetcode.com/problems/divide-two-integers/
```
class Solution:
    def divide(self, dividend: int, divisor: int) -> int:
        a = abs(dividend)
        b = abs(divisor) 
        is_neg = -1 if (a == dividend) ^ (b == divisor) else 1 
        q = 0
        
        while a >= b:
            temp_q = 1
            temp_b = b
            while (temp_b + temp_b) < a:
                temp_b += temp_b
                temp_q += temp_q
            q += temp_q
            a -= temp_b 
            
        return min(2**31-1, q*is_neg)
