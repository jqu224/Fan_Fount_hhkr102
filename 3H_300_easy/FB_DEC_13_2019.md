91. Decode Ways
https://leetcode.com/problems/decode-ways/
class Solution:
    def numDecodings(self, s: str) -> int:
        if s[0] == "0":
            return 0
        prev = ""
        memo = [1]
        
        for i in s:
            if prev and i == "0":
                memo.append(memo[-2])
            elif prev == i == "0":
                return 0
            elif prev and prev in "12" and i in "0123456": 
                memo.append(memo[-2] + memo[-1])
            else:
                memo.append(memo[-1])
            prev = i 
        return memo[-1]
                
                
                
class Solution:
    def numDecodings(self, s: str) -> int:
        if len(s) == 0:
            return 0
        if s[0] == "0":
            return 0
        dp = [0] * (len(s) + 1)
        dp[0] = 1
        dp[1] = 1
        for i in range(2, len(s)+1):
            number = int(s[i-1])
            # print(number)
            nb_with_prev = int(s[i-2:i])
            # print(nb_with_prev)
            if nb_with_prev <= 26 and nb_with_prev > 9:
                dp[i] += dp[i-2]
            if number > 0:
                dp[i] += dp[i-1]
            if dp[i] == 0:
                return 0
        return dp[-1]
